create table Ingredients (id INT AUTO_INCREMENT PRIMARY KEY, name VARCHAR(255) NOT NULL, amount VARCHAR(255) NOT NULL, unit VARCHAR(255) NOT NULL);
create table Meals (id INT AUTO_INCREMENT PRIMARY KEY, name VARCHAR(255) NOT NULL, link VARCHAR(255) NOT NULL, category VARCHAR(255) NOT NULL, instruction VARCHAR(255) NOT NULL);
create table MealPlans (id INT AUTO_INCREMENT PRIMARY KEY, name VARCHAR(255) NOT NULL);
